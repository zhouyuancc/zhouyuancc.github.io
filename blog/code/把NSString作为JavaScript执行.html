/**

 网页加载完毕的时候调用

 */

- (void)webViewDidFinishLoad:(UIWebView *)webView

{

    //跳到id对应的网页标签

//    NSLog(@"%@",self.html.ID);

    

    //1.拼接JavaScript代码

    NSString *js = [NSString stringWithFormat:@"window.location.href = '#%@';",self.html.ID];

    //Evaluating执行

    //把NSString作为JavaScript执行

    

    //2.执行JavaScript代码

    [webView stringByEvaluatingJavaScriptFromString:js];

}
