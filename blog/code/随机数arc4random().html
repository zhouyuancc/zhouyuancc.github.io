/**

 随机选中某种食物

 */

- (IBAction)randomFood {

    //1000%12 -> 0-11 数组有11条数据

    //任意整数 膜 12

    //arc4random()%12

    //self.foods[0].count //[self.foods[0] count]

    for(int i = 0; i < self.foods.count ; i++)

    {

//        NSLog(@"%d",i);

        long count = [self.foods[i] count];//第i列数组 数据个数

   

        long oldRow = [self.pickerView numberOfRowsInComponent:i];//当前pickerView 第i列的行号


        long row = oldRow;

        

        //保证行数跟上一次不同

        while (row == oldRow) {//不用if用while 循环多次

            row = arc4random()%count;//随机数%count

  //        row = arc4random_uniform(count);//等价

        }

        

//        NSLog(@"%ld",[self pickerView:nil numberOfRowsInComponent:0]);

        

        [self.pickerView selectRow:row inComponent:i animated:YES];

        

        //设置lable的文字

        [self pickerView:nil didSelectRow:row inComponent:i];

    }

}
