/**

 将分号隔开的中文姓名 转换为 分号隔开的英文姓名

 */

-(NSString *)getEmpCodesByEmpNames:(NSString *)EmpNames

{

    NSArray * NamesArray = [EmpNames componentsSeparatedByString:@";"];

    NSMutableString *result = [NSMutableString stringWithCapacity:[NamesArray count]];

    AddRessbookAppdataStore *addrStore = [AddRessbookAppdataStore sharedStore];

    for(NSString *name in NamesArray)

    {

        NSString *code = [addrStore getEmpCodeByRealName:name];

        [result appendString:[NSString stringWithFormat:@"%@;",code]];

    }

    return  result;

}
