- (IBAction)save {

    //快速存储 本质也是plist存储

    

    //1.利用NSUserDefaults 就能直接访问软件的偏好设置（Library\Preferences）

    NSUserDefaults *defaults = [NSUserDefaults standardUserDefaults];

    

    //2.存储数据

    [defaults setObject:@"zhouyuan" forKey:@"account"];

    [defaults setObject:@"1" forKey:@"pwd"];

    [defaults setInteger:20 forKey:@"age"];

    [defaults setBool:YES forKey:@"auto_login"];

    

//    NSDictionary *dict = @{@"account":@"zhouyuan",

//                           @"auto_login":@YES,

//                           @"age":@10

//                           };

    

    //3.立刻同步 synchronize:同步

    [defaults synchronize];    

}


- (IBAction)read {

    NSUserDefaults *defaults = [NSUserDefaults standardUserDefaults];

    

    NSString *account = [defaults objectForKey:@"account"];

    BOOL autoLogin = [defaults boolForKey:@"auto_login"];

    NSLog(@"%@---%d",account,autoLogin);

}
