/**

 scrollview滚动到底部

 */

- (void)scrollsToBottomAnimated:(BOOL)animated

{

    CGFloat offset = self.scrollView.contentSize.height - self.scrollView.bounds.size.height;

    NSLog(@"%f",self.scrollView.contentSize.height);

    NSLog(@"%f",self.scrollView.bounds.size.height);

    

    if (offset > 0)

    {

        [self.scrollView setContentOffset:CGPointMake(0, offset) animated:animated];

    }

}
