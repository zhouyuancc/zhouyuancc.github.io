/**

* 特殊窗体 

* iframe/弹窗/对话框 

* switchTo()

*/

public void iframeSwitch() {


// 富文本编辑器

// iframe获取id

WebElement iframeElement = driver.findElement(By.id("ueditor_0"));


// 切换到富文本

// 富文本里面是另一个<html><body></body></html>

driver.switchTo().frame(iframeElement);


// body中输入文字

driver.findElement(By.tagName("body")).sendKeys("test");

}



public String windowsHandle;// 当前window


/**

* 窗体间切换: 所有getWindowHandles()/当前getWindowHandle()

*/

public void windowsHandleAction() {

try {


sleep();


// 获取左侧菜单

WebElement leftElement = driver.findElement(By.className("scroll-left-fat"));

// 获取[工作流]模块

WebElement workFlowElement = leftElement.findElement(By.linkText("工作流"));

// 鼠标悬浮在[工作流]上

Actions actions = new Actions(driver);

actions.moveToElement(workFlowElement).perform();// 必须加perform()

// 点击[我的工作]

WebElement aElement = leftElement.findElement(By.partialLinkText("我的工作"));

aElement.click();


sleep();


//注意: 按层级获取iframe会报错, stale: either the element is no longer attached to the DOM or the page has been refreshed -> 元素不再是附加到DOM或者页面被刷新

//获取[我的工作]右侧结果列表中 

//div -> iframe: (id)appiframe_2 -> iframe: (id)main -> a:表单名

WebElement element =driver.switchTo().frame("appiframe_2").switchTo().frame("main").findElement(By.partialLinkText("请假申请单"));

//点击流程名称 -> 弹出新页面

element.click();

sleep();


// 获取当前handle

getCurrentWindowsHandle();


// 拿到所有handle窗口

Set<String> handles = driver.getWindowHandles();

System.out.println("handles: " + handles);

//driver切换到新弹出的窗口

for (String s : handles) {

if (s.equals(windowsHandle)) {

continue;// 跳出当前循环

}


System.out.println(s);

//driver切换到新弹出的窗口

driver.switchTo().window(s);

}


// 修改弹出的表单上的请假天数

WebElement daysOffElement = driver.findElement(By.name("DATA_14"));

daysOffElement.clear();

daysOffElement.sendKeys("5");


} catch (Exception e) {

// TODO: handle exception

System.out.println(e);

driver.quit();

}

}


// 获取当前窗口

public void getCurrentWindowsHandle() 

{

// 获取当前窗口

windowsHandle = driver.getWindowHandle();

System.out.println("首页-当前窗口: " + windowsHandle);

}
