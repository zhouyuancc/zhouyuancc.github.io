/**

* input文本框 

* sendkeys 

* clear 

* getAttribute

*/

public void inputBox() {


// 清空 用户名文本框 

driver.findElement(By.name("email")).clear();

 

// 获取 属性名 为 placeholder 的值 

zyGetAttribute();

 

// 登录

// 获取到用户名文本框, 输入

driver.findElement(By.name("email")).sendKeys("zhouyuancc@gmail.com");

// 获取到密码文本框, 输入

driver.findElement(By.name("password")).sendKeys(“123456”);

// 点击登录

driver.findElement(By.xpath(".//*[@id='signup-form']/div[5]/input")).click();

}


/**

* 获取 属性名 为 placeholder 的值

*/

public void zyGetAttribute() {


//<input id="email" placeholder="用户名" type="text">


// 获取 属性名 为 placeholder 的值

String placeholderStr = driver.findElement(By.name("email")).getAttribute("placeholder");

System.out.println(placeholderStr);

}
