// 封装基本方法

/**

* 获取父element

*/

public WebElement getElement(By by) {

return driver.findElement(by);

}


/**

* 通过父节点获取子节点

*/

public WebElement getElement(WebElement element, By by) {

return element.findElement(by);

}


/**

* 获取商品文本信息

*/

public String getCourseText(WebElement element) {

return element.getText();

}


// 封装购买方法

/**

* 1.立即购买, 返回购买信息

* 

* @throws Exception

*/

public String buyCourseNow() throws Exception {


WebElement element = this.getElement(GetByLocator.getLocator("courseName"));

WebElement elementNode = this.getElement(element, GetByLocator.getLocator("courseNameNode"));

String currentText = this.getCourseText(elementNode);


// 点击立即购买

driver.click(this.getElement(GetByLocator.getLocator("buyNow")));


return currentText;

}


/**

* 2.点击提交订单

* 

* @throws Exception

*/

public void sureOrder() throws Exception {

driver.click(this.getElement(GetByLocator.getLocator("submitOrder")));

}


/**

* 3.获取订单号

* 

* @throws Exception

*/

public String getOrder() throws Exception {

String orderText = this.getCourseText(this.getElement(GetByLocator.getLocator("orderText")));

return orderText;

}


/**

* 4.获取支付中心商品文本信息

* 

* @throws Exception

*/

public String getOrderCourse() throws Exception {

WebElement element = this.getElement(GetByLocator.getLocator("courseName2"));

WebElement elementNode = this.getElement(element, GetByLocator.getLocator("courseNameNode2"));

return this.getCourseText(elementNode);

}
