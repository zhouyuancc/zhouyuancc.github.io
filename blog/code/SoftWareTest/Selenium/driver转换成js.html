/**

* driver转换成js

*/

public void driverToJs(String jsString) {


// 控制台输入

// a标签获取元素: document.getElementsByClassName('js-avator-link')

// a标签元素赋值:

// document.getElementsByClassName('js-avator-link')[0].style.bottom='0';


// 赋值后: js-avator-link是a标签的class, 无效, 因为上级div也不可见

// 需使用上级div的class: update-avator, 其bottom也是-30px


// 更换 头像按钮 本是bottom:-30px不可见的

// driver转换成js

// String jsString = "document.getElementsByClassName('update-avator')[0].style.bottom='0';";


// driver转js JavascriptExecutor

JavascriptExecutor js = (JavascriptExecutor) driver;

// 执行js

js.executeScript(jsString);


}


// 调用js,令隐藏的 更换头像 按钮,展示

driverToJs("document.getElementsByClassName('update-avator')[0].style.bottom='0';");


// <input id="upload" class="hide" title="上传头像" name="fileField" accept="image/gif,image/jpeg,image/jpg,image/png" type="file">

//展示上传浏览控件

driverToJs("document.getElementById('upload').className='display';");
