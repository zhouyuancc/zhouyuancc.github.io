/**

* radio单选框 

* click 

* clear 

* isSelected

*/

public void radioBox() 

{

// <div class="moco-control-input rlf-group rlf-radio-group">

// <label>

// <input hidefocus="true" value="0" name="sex" type="radio">

// 保密

// </label>

// <label>

// <input hidefocus="true" value="1" name="sex" type="radio">

// 男

// </label>

// <label>

// <input hidefocus="true" value="2" checked="checked" name="sex" type="radio">

// 女

// </label>

// </div>


// 单选框内容: 保密\男\女

// 点击[女]单选框

driver.findElement(By.xpath(".//*[@id='profile']/div[4]/div/label[3]/input")).click();


// label//input: 获取所有label下的单选框元素

List<WebElement> elements = driver.findElements(By.xpath(".//*[@id='profile']/div[4]/div/label//input"));


// 打印单选按钮个数

System.out.println(elements.size());


// 遍历单选按钮

for (WebElement radio : elements) {

// 判断该单选按钮是否被选中

boolean flag = radio.isSelected();

// 未被选中

if (flag == false) {

// 点击radio

radio.click();

break;

} else {

// 已被选中

System.out.println("单选框" + radio.getText() + "已选中");// 保密\男\女

}

}

}
