/**

* 下拉框 

* 选择对应的元素: Index/Value/Text 

* 不选择对应的元素: deselectAll/deselectByValue/deselectByVisibleText 

* 获取选择项的值: getAllSelectedOptions()/getFirstSelectedOption().getText

*/

public void select() 

{

// 下拉操作

// ElementNotInteractableException

// 直接获取job:获取不到元素

// job不止一个

// 难点: 弹窗上的控件需层级定位,通过form表单查找

// WebElement job = driver.findElement(By.id("job”));//ElementNotInteractableException


// 一级一级进行元素获取

//先获取form表单

WebElement formElement = driver.findElement(By.id("profile"));

//再获取select下拉框

WebElement job = formElement.findElement(By.id("job"));


/**

* <select id="job" class="moco-form-control rlf-select" name="job" hidefocus="true" data-validate="require-select"> 

* <option value="">请选择职位</option>

* <option value="13">学生</option>

* <option value="7" selected="selected">移动开发工程师</option>

* <option value="9">软件测试工程师</option> </select>

*/


/**

* 下拉框

*/

Select downList = new Select(job);

/**

* 选中

*/

// selectBy(downList);


downList.selectByIndex(1);


/**

* 不选中

*/

// deselectBy(downList);


/**

* 操作get

*/

getSelectOptions(downList);

}


/**

* 选中: selectByIndex/selectByValue/selectByVisibleText

* 下拉框: Select downList = new Select(job);

* @param downList

*/

public void selectBy(Select downList) {


// Select downList = new Select(job);

/**

* 选中

*/

// 根据index 

downList.selectByIndex(0);//<option value="">请选择职位</option>


sleep();

// 根据value 

downList.selectByValue("7");//<option value="7">


sleep();

// 根据可见文字 

downList.selectByVisibleText("软件测试工程师");//<option>软件测试工程师</option>

}

/**

* 操作: getAllSelectedOptions().get(0)/getFirstSelectedOption().getText()/getOptions()

* 下拉框: Select downList = new Select(job); 

* @param downList

*/

public void getSelectOptions(Select downList) 

{

// 获取所有选中的下拉框数据的第一个: getAllSelectedOptions().get(0)

System.out.println(downList.getAllSelectedOptions().get(0));

// [[[[[[FirefoxDriver: firefox on MAC (3603f70d-7180-3e4f-8743-5653b113eaf7)] -> id: profile]] -> id: job]] -> tag name: option]


// 获取选中的第一个下拉框数据的文字: getFirstSelectedOption().getText()

System.out.println(downList.getFirstSelectedOption().getText());


// 获取所有选中的下拉框数据: getAllSelectedOptions

List<WebElement> selectedOptions = downList.getAllSelectedOptions();

//遍历

for (WebElement webElement : selectedOptions) {

//打印多选框文本

System.out.println(webElement.getText());

}


// 获取所有下拉框数据: getOptions

List<WebElement> options = downList.getOptions();

for (WebElement webElement : options) {

//打印多选框文本

System.out.println(webElement.getText());

}

}
