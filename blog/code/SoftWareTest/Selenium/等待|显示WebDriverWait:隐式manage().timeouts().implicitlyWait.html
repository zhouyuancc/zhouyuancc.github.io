/**

* 等待

* 强制等待: Thread.sleep(2000)

* 显示等待: WebDriverWait|时间内找到该元素,继续执行,若没找到,报错; 何时找到就直接执行下去

* 隐式等待: manage().timeouts().implicitlyWait|全局, 10s内每个元素都去找,找到了再执行下一步//若是测inputBox(); 需把该方法写在其前面//调试时常用

*/

public void waitFunction() {


// 强制等待

sleep();

sleepWithTime(2000);


// 显示等待

// 时间内找到该元素,继续执行,若没找到,报错; 何时找到就直接执行下去

new WebDriverWait(driver, 15).until(ExpectedConditions.presenceOfElementLocated(By.cssSelector("css locator")));


// 隐式等待

// 只是调试时常用

// 全局, 10s内每个元素都去找,找到了再执行下一步// 写在其他方法前面

// 若是测inputBox(); 需把该方法写在其前面

// aSelenium.waitElement();

// aSelenium.inputBox();

driver.manage().timeouts().implicitlyWait(10, TimeUnit.SECONDS);

}


/**

* 显示等待

*/

public void waitElement() {


// 显示等待

WebDriverWait wait = new WebDriverWait(driver, 10);

// 在当前页面定位元素, 10s内未找到会报错

wait.until(ExpectedConditions.presenceOfElementLocated(By.id("test")));

}

*/

public void sleep() {

try {

Thread.sleep(2000);

} catch (InterruptedException e) {

}

}


/**

* 传入等待时间

* @param time

*/

public void sleepWithTime(long time) {

try {

Thread.sleep(time);

} catch (InterruptedException e) {

// TODO: handle exception

e.printStackTrace();

}

}
