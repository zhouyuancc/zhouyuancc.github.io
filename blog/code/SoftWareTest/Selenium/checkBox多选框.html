/**

* checkBox多选框 

* click 

* clear 

* isSelected 

* isEnabled

*/

public void checkBox() {


// 获取登录页面的自动登录多选框

WebElement checkElement = driver.findElement(By.id("auto-signin"));


// checkBox是否选中

boolean checkIsSelected = checkElement.isSelected();

// checkBox是否可用

boolean checkIsEnabled = checkElement.isEnabled();


System.out.println("checkBox是否选中:" + checkIsSelected);

System.out.println("checkBox是否可用:" + checkIsEnabled);


// clear取消checkBox选中,无用,会报错

// checkElement.clear();


if (!checkIsSelected) // 未选中

{

// 未选中时,点击checkBox,令其选中

checkElement.click();

}

}
