/**

* 鼠标操作

* 左击: actions.click(webElement).perform();

* 右击: actions.contextClick(webElement).perform();

* 双击: actions.doubleClick(webElement).perform(); // 不好用

*/

public void mouseActions() {

/**

* 左击/右击/双击

*/

//获取用来打开登录界面的按钮

WebElement login = driver.findElement(By.id("js-signin-btn"));

 

//打开登录页面 

Actions actions = new Actions(driver); 

//提交: perform() | 不加不能用 

//单击

actions.click(login).perform();

 

sleep(); 

//双击 

actions.doubleClick(login).perform();//不好用

 

sleep(); 

//右击: contextClick 

actions.contextClick(login).perform();

}


/**

* 鼠标操作

* 悬停: actions.moveToElement(webElement).perform()

*/

public void mouseMoveTo() {


//   <div class="menuContent"> 

//   <div class="item" data-id="a">

// <a href="/course/list?c=fe" target="_blank"> 

// <span class="group">工作流</span>

// <i class="icon-right2"></i> 

//     </a> 

//   </div> 

//         <div class="item" data-id="b">

//   ...

//         </div>

//   </div>

// 获取 大的div

WebElement webElement = driver.findElement(By.className("menuContent"));

// 通过 大的div, 获取class="item"的List数组

List<WebElement> itemList = webElement.findElements(By.className("item"));


Actions actions = new Actions(driver);

// 鼠标移动到itemList的第一个元素上, perform提交

actions.moveToElement(itemList.get(0)).perform();


sleep();


// <a>我的工作</a>

// 链接上展示的文字: linkText

driver.findElement(By.linkText("我的工作")).click();

}
