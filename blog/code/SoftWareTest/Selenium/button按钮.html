/**

* button按钮 

* click 

* isEnabled

*/

public void button() {


// 通过class="btn-red btn-full xa-login"获取登录按钮

WebElement loginBtn = driver.findElement(By.className("btn-red"));

// 打印登录按钮的value属性

System.out.println(loginBtn.getAttribute("value"));


// button是否可用

boolean loginBtnIsEnable = loginBtn.isEnabled();

System.out.println(loginBtnIsEnable);


// 点击登录

loginBtn.click();

}
