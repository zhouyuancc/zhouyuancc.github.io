public void deleteCookie() {

从cookie.properties配置文件中,获取添加cookie/写入cookie

package com.SG.ZY.util;


import java.io.IOException;

import java.util.Set;


import org.openqa.selenium.Cookie;


import com.SG.ZY.base.BaseDriver;


/**

 * 从cookie.properties配置文件中,获取添加cookie/写入cookie

 * @author Zhou

 *

 */

public class HandleCookie {


public BaseDriver driver;

public ProUtil proUtil;

public HandleCookie(BaseDriver driver) throws IOException {

// TODO Auto-generated constructor stub

this.driver = driver;

this.proUtil = new ProUtil("cookie.properties");

}

/**

* 从cookie.properties配置文件读取并添加cookie

* @throws Exception

*/

public void setCookie() throws Exception {


String key = "apsid";

String value = proUtil.getPro(key);

//根据Cookie.properties创建Cookie

Cookie cookie = new Cookie(key, value, "imooc.com", "/", null);//domain : 域名 , "/" : 全部路径下

//设置cookie

this.driver.setCookie(cookie);

}

/**

* cookie.properties配置文件写入cookie

* @throws Exception

*/

public void writeCookie() throws Exception {

String key = "apsid";

//从网页获取cookie

Set<Cookie> cookies = this.driver.getCookie();

for (Cookie cookie : cookies) {

if (cookie.getName().equals(key)) {

//写入配置文件cookie.properties

proUtil.writePro(cookie.getName(), cookie.getValue());

}

}

}

}
