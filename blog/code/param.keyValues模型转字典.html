@implementation ZYStatusTool


+(void)newStatusWithSinceId:(NSString *)sinceId success:(void (^)(NSArray *))success failure:(void (^)(NSError *))failure

{

    //创建参数模型

    ZYStatusParam *param = [[ZYStatusParam alloc] init];

    param.access_token = [ZYAccountTool account].access_token;

    

    //创建一个参数字典

    if(sinceId)

    {

        param.since_id = sinceId;

    }

    /**

     success:^(id responseObject) {

     //请求成功的回调

     }

     */

    void (^successHttp)(id) = ^(id responseObject) {

        /***********************/

        /**

         HttpTool请求成功 后 的 回调

         */

        /***********************/

        //请求成功代码 先保存

        

        //获取到微博数据 转换成模型

        //获取微博字典数组

        NSArray *dictArr = responseObject[@"statuses"];

        /**

         把字典数组转换成模型那个数组

         */

        NSArray *statuses = (NSMutableArray *)[ZYStatus objectArrayWithKeyValuesArray:dictArr];

        

        if(success)

        {

            success(statuses);

        }

    };

————————————————————————————————————————————————————————

    /**

     param.keyValues模型转字典

     */

————————————————————————————————————————————————————————

    /**

     发送get请求

     */

————————————————————————————————————————————————————————

    [ZYHttpTool GET:@"https://api.weibo.com/2/statuses/friends_timeline.json" parameters:param.keyValues success:successHttp failure:^(NSError *error) {

————————————————————————————————————————————————————————

        if (failure) {

            failure(error);

        }


    }];

    

}


@end
