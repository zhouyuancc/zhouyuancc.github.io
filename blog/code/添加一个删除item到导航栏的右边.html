- (void)viewDidLoad {

    [super viewDidLoad];

    

    //无分割线

    self.tableView.separatorStyle = UITableViewCellSeparatorStyleNone;


    //有数据那一行有分割线

    //正常

//    self.tableView.separatorStyle = UITableViewCellSeparatorStyleSingleLine;

    //添加1条数据，就不显示分割线

//    self.tableView.separatorStyle = UITableViewCellSeparatorStyleSingleLineEtched;

    

    //添加一个删除item到导航栏的右边

//    NSLog(@"%@",self.navigationItem.rightBarButtonItem);

    UIBarButtonItem *add = self.navigationItem.rightBarButtonItem;

    UIBarButtonItem *deleteItem = [[UIBarButtonItem alloc] initWithBarButtonSystemItem:UIBarButtonSystemItemTrash target:self action:@selector(deleteClick)];

    //rightBarButtonItems

    self.navigationItem.rightBarButtonItems = @[add,deleteItem];

}


- (void)deleteClick

{

    //让tableView进入编辑状态

//    self.tableView.editing = YES;

    //取反

//    self.tableView.editing = !self.tableView.isEditing;

    //加动画

    [self.tableView setEditing:!self.tableView.isEditing animated:YES];

}
