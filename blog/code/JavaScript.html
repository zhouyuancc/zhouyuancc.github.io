DOM操作-图片上一张下一张切换


<img class="icon" src="image/icon_01.png">

<p></p>

<button class="pre">上一张</button>

<button class="next">下一张</button>


<script src="js/index.js"></script>

<script type="text/javascript">


    //拿到所有的标签

    var icon = document.getElementsByClassName('icon')[0];

    var btnPre = document.getElementsByClassName('pre')[0];

    var btnNext = document.getElementsByClassName('next')[0];


    var maxImgCount = 9;

    var minImgCount = 1;

    var currentImg = minImgCount;


    //点击上一张

    btnPre.onclick = function(){


        if(currentImg == minImgCount)

        {

            currentImg = maxImgCount;

        }

        else

        {

            currentImg = currentImg - 1;

        }

        //!拼接

        icon.src = 'image/icon_0' + currentImg + '.png';

    }


    //点击下一张

    btnNext.onclick = function(){


        if(currentImg == maxImgCount)

        {

            currentImg = minImgCount;

        }

        else

        {

            currentImg = currentImg + 1;

        }

        //!拼接

        icon.src = 'image/icon_0' + currentImg + '.png';

    }

</script>
