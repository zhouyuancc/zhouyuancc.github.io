#import <UIKit/UIKit.h>


@class FoodTypesViewController,FoodType;


@protocol FoodTypesViewControllerDelegate<NSObject>

@optional

- (void)foodTypesViewController:(FoodTypesViewController *)foodTypesVc didSelectedFoodType:(FoodType *)type;


@end


@interface FoodTypesViewController : UITableViewController


@property (nonatomic,weak) id<FoodTypesViewControllerDelegate> delegate;


@end


//

//  FoodTypesViewController.m

//  02-UISplitViewController

//

//  Created by ZhouYuan on 16/2/1.

//  Copyright © 2016年 ZhouYuan. All rights reserved.

//

- (void)tableView:(UITableView *)tableView didSelectRowAtIndexPath:(NSIndexPath *)indexPath

{

    if ([self.delegate respondsToSelector:@selector(foodTypesViewController:didSelectedFoodType:)]) {

        

        FoodType *foodType = self.foodTypes[indexPath.row];


        [self.delegate foodTypesViewController:self didSelectedFoodType:foodType];

    }

}
