- (void)getV

{

    if (![self.phone.text isEqualToString:nil]) {

        self.phoneNumber.text = self.phone.text;

    }

    [self.phone setText:nil];

    self.phone.placeholder = @"请输入验证码";

    self.secondsCountDown = 60;

    self.cue.text = [NSString stringWithFormat:@"%d",self.secondsCountDown];

    /**

     每秒执行一次

     */

    NSTimer *timer = [NSTimer scheduledTimerWithTimeInterval:1.0 target:self selector:@selector(timeCountDown) userInfo:nil repeats:YES];

    self.timer = timer;

    

}


- (void)timeCountDown

{

    self.secondsCountDown --;

    self.cue.text = [NSString stringWithFormat:@"%d",self.secondsCountDown];

    if (self.secondsCountDown == 0) {

        [self.timer invalidate];

        self.cue.text = nil;

        [self.getVBtn setTitle:@"重新获取验证码" forState:UIControlStateNormal];

    }


}
