@interface EditEmailTableViewController ()<UIAlertViewDelegate>

//人员按钮

@property (nonatomic,strong) EmpButton *empButton;


@end



- (void)editTableViewCellDidClickEmpBtn:(EmpButton *)button

{

//    NSLog(@"editTableViewCellDidClickEmpBtn %@",button.name);

    

    self.empButton = button;

    

    NSString *message = [NSString stringWithFormat:@"您确定要删除收件人%@吗？",button.name];

    UIAlertView *alert = [[UIAlertView alloc] initWithTitle:nil message:message delegate:self cancelButtonTitle:@"取消" otherButtonTitles:@"确定", nil];

    [alert show];

    

}


#pragma mark - alertView代理方法

- (void)alertView:(UIAlertView *)alertView clickedButtonAtIndex:(NSInteger)buttonIndex

{

    if(buttonIndex == 0)

        return;

    

    //确定按钮


    //拼接字符串

    NSString *replaceStrCode = [self.empButton.code stringByAppendingString:@","];//要删除的编号

    NSString *replaceStrName = [self.empButton.name stringByAppendingString:@","];//要删除的姓名

    

    EditEmailFrame *editFrame = self.editEmailFrame[0];

    NSString *Code = editFrame.editEmail.code;//收件人 原始所有编号

    NSString *Name = editFrame.editEmail.content;//收件人 原始所有姓名


//    NSLog(@"replaceStrCode %@ name %@",replaceStrCode,replaceStrName);

//    NSLog(@"%@ %@",Code,Name);

    

    //删除后的收件人

    //替换

    NSString *currentCode = [Code stringByReplacingOccurrencesOfString:replaceStrCode withString:@""];

    NSString *currentName = [Name stringByReplacingOccurrencesOfString:replaceStrName withString:@""];

    

    

//    NSLog(@"currentCode %@  currentName %@",currentCode,currentName);

    

    [self reload:currentCode WithName:currentName];


}


/**

 刷新数据 收件人

 //全局刷新

 */

- (void)reload:(NSString *)code WithName:(NSString *)name

{

    EditEmailFrame *edit = self.editEmailFrame[0];


    EditEmail *editEmail = [[EditEmail alloc] init];

    editEmail.name = @"收件人:";

    editEmail.content = name;//中文

    editEmail.code = code;//编号

    editEmail.add = YES;

    editEmail.unEnable = YES;//不可写

    

    edit.cellHeight = [edit changeCellHeight:editEmail];

    

    //        NSLog(@"%f",edit.cellHeight);


    //全局刷新

    [self.tableview reloadData];

    

    //因为第一行cellHeight行高改变后，其他行的尺寸也有变动，所以此时不用局部刷新

    //        NSIndexPath *path = [NSIndexPath indexPathForRow:0 inSection:0];

    //        [self.tableview reloadRowsAtIndexPaths:@[path]     withRowAnimation:UITableViewRowAnimationBottom];

}
