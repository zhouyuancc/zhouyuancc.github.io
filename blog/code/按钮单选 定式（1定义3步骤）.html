@property (nonatomic,weak) ZYSegment *selectedButton;


- (void)buttonClick:(ZYSegment *)button

{

    self.selectedButton.selected = NO;

    //不要忘了这句

    button.selected = YES;

    

    self.selectedButton = button;

}
