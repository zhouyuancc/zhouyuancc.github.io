#pragma mark - 计算微博配图的尺寸 公式：总行数 = （总个数 - 1）/ 总列数 + 1

- (CGSize)photosSizeWithCount:(int)count

{

    //5

    //1 2 3

    //4 5

    

    //4幅图 2行2列

    //其他  n行3列

    

    //count : 图片总数

——————————————————————————————————————————————————————————————————————————————————————————————————————    

    //获取总列数

    int cols = count == 4?2:3;

    

    //获取总行数 5

    /**

     总行数 = （总个数 - 1）/ 总列数 + 1

     */

    int rols = (count - 1) / cols + 1;

    

    CGFloat photoWH = 70;

    CGFloat w = cols * photoWH + (cols - 1) * ZYStatusCellMargin;

    CGFloat h = rols * photoWH + (rols - 1) * ZYStatusCellMargin;

——————————————————————————————————————————————————————————————————————————————————————————————————————

    

    return CGSizeMake(w, h);

}

#import "ZYPhotosView.h"

#import "ZYPhoto.h"

#import "UIImageView+WebCache.h"


@implementation ZYPhotosView


- (instancetype)initWithFrame:(CGRect)frame

{

    if (self = [super initWithFrame:frame]) {

//        self.backgroundColor = [UIColor redColor];

        

        //添加9个子控件

        [self setUpAllChildView];

        

    }

    return self;

}


/**

 添加9个子控件

 */

- (void)setUpAllChildView

{

    for (int i = 0; i < 9; i++) {

        UIImageView *imageV = [[UIImageView alloc] init];

——————————————————————————————————————————————————————————————————————————————————————————————————————

        /**

         UIViewContentModeScaleToFill 填充整个控件

         

         UIViewContentModeScaleAspectFit 按比例缩放，并且宽高！都不！超出控件

         UIViewContentModeScaleAspectFill 按比例缩放，直到！有一个！宽度或高度不超过控件就可以，自动调节中心点

         

         UIViewContentModeCenter 不会缩放图片，只会把图片的中心点对齐控件

         */

        imageV.contentMode = UIViewContentModeScaleAspectFill;

        

        //裁剪图片，超出控件的部分裁剪掉

        imageV.clipsToBounds = YES;

——————————————————————————————————————————————————————————————————————————————————————————————————————        

        [self addSubview:imageV];

    }

}


- (void)setPic_urls:(NSArray *)pic_urls

{

    _pic_urls = pic_urls;

    

    int count = (int)self.subviews.count;

    for(int i = 0;i < count;i++)

    {

        UIImageView *imageV = self.subviews[i];

        

        if (i < _pic_urls.count) {//显示

            

            imageV.hidden = NO;

            

            //获取ZYPhoto模型

            ZYPhoto *photo = _pic_urls[i];

——————————————————————————————————————————————————————————————————————————————————————————————————————            

            //赋值

            [imageV sd_setImageWithURL:photo.thumbnail_pic placeholderImage:[UIImage imageNamed:@"timeline_image_placeholder"]];

——————————————————————————————————————————————————————————————————————————————————————————————————————

        }

        else

        {

            imageV.hidden = YES;

        }

    }

}


/**

 计算尺寸

 */

- (void)layoutSubviews

{

    [super layoutSubviews];

    

    CGFloat x = 0;

    CGFloat y = 0;

    CGFloat w = 70;

    CGFloat h = 70;

    CGFloat margin = 10;

    

    int col = 0;

    int rol = 0;

    //总列数

    int cols = _pic_urls.count == 4?2:3;

    

    //计算显示出来的imageView

    for (int i = 0; i < _pic_urls.count; i++) {

——————————————————————————————————————————————————————————————————————————————————————————————————————     

        //当前图片所在的列数

        col = i % cols;

        //当前图片所在的行数

        rol = i /cols;

——————————————————————————————————————————————————————————————————————————————————————————————————————       

        UIImageView *imageV = self.subviews[i];

        x = col * (w + margin);

        y = rol * (h + margin);

        imageV.frame = CGRectMake(x, y, w, h);

    }

    

}


@end
