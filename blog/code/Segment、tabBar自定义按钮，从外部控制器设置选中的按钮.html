/**

 */

- (void)setupZYSegmentControl

{

    //设置控制器顶部的控件

    ZYSegmentControl *sc = [[ZYSegmentControl alloc] init];

    sc.items = @[@"全部",@"特别关心",@"好友动态",@"认证空间"];

    //整个titleView的尺寸

    sc.frame = CGRectMake(0, 0, 400, 30);

    //默认选中第0个

    sc.selectedSegmentIndex = 0;

    self.navigationItem.titleView = sc;

}

//UIButton

@property (nonatomic,weak) ZYSegment *selectedSegment;


//点击事件

[segment addTarget:self action:@selector(segmentClick:) forControlEvents:UIControlEventTouchDown];


[self addSubview:segment];


/**

 单选

 */

- (void)segmentClick:(ZYSegment *)segment

{

    self.selectedSegment.selected = NO;

    //不要忘了这句

    segment.selected = YES;

    

    self.selectedSegment = segment;

}


/**

 set

 */

- (void)setSelectedSegmentIndex:(NSInteger)selectedSegmentIndex

{

    NSInteger count = self.subviews.count;

    if(selectedSegmentIndex < 0 || selectedSegmentIndex >= count) return;

    

    ZYSegment *segment = self.subviews[selectedSegmentIndex];

    [self segmentClick:segment];

}

/**

 get

 */

- (NSInteger)selectedSegmentIndex

{

    return self.selectedSegment.tag;

}
