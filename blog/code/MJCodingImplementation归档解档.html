#import "ZYAccountTool.h"

#import "ZYAccount.h"


//#define ZYAccountFileName [NSSearchPathForDirectoriesInDomains(!!NSDocumentationDirectory!!, NSUserDomainMask, YES)[0] stringByAppendingPathComponent:@"account.data"]


#define ZYAccountFileName [NSSearchPathForDirectoriesInDomains(NSDocumentDirectory, NSUserDomainMask, YES)[0] stringByAppendingPathComponent:@"account.data"]


@implementation ZYAccountTool


—————————————————————————————

/**

 类方法一般用静态变量代替成员属性

 */

static ZYAccount *_account;

—————————————————————————————


/**

 archiver

 文档，归档器

 n. 存储服务器，压缩程式；文档，静态无图文档

 */


+ (void)saveAccount:(ZYAccount *)account

{

    [NSKeyedArchiver archiveRootObject:account toFile:ZYAccountFileName];

}


+ (ZYAccount *)account

{

    if (_account == nil) {

        

       _account = [NSKeyedUnarchiver unarchiveObjectWithFile:ZYAccountFileName];

        

        //过期时间 = 当前保存时间 + 有效期

        //判断账号是否过期，如果过期直接返回Nil

        //获取当前时间 < 过期时间

        if ([[NSDate date] compare:_account.expires_date] != NSOrderedAscending) {//过期//非递增

            return nil;

        }

        

    }

    

    return _account;

}
