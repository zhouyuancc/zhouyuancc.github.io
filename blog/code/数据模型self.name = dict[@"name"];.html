- (instancetype)initWithDict:(NSDictionary *)dict

{

    if(self = [super init])//自定义构造方法 固定写法

    {

        self.icon = dict[@"icon"];

        self.name = dict[@"name"];

    }

    return self;

}


+ (instancetype)appWithDict:(NSDictionary *)dict

{    

    return [[self alloc] initWithDict:dict];

}
