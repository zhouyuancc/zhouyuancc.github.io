#pragma mark - 判断初次进入页面时是否自动刷新

- (BOOL)isRefreshDataWithRefreshTime:(NSString *)refreshTime

{

    //获取当前时间

    NSString *nowTime = [AppdataStore getNowDateClass];


    //获取最新刷新时间

    int compare = [self compareDate:nowTime withDate:refreshTime];

    

    //    NSLog(@"nowTime %@,refreshTime %@",nowTime,refreshTime);

    

    //3小时

    //    compare = [comDate compareDate:@"2015-10-22 13:12:11" withDate:@"2015-10-22 10:12:11"];

    if (compare > 10800.00) {

        return YES;

    }

    return NO;

}

/**

 比较两日期大小

 */

-(double)compareDate:(NSString *)date01 withDate:(NSString *)date02

{

    NSDateFormatter *df = [[NSDateFormatter alloc] init];

    [df setDateFormat:@"yyyy-MM-dd HH:mm:ss"];

    

    NSDate *dt1 = [[NSDate alloc] init];

    dt1 = [df dateFromString:date01];

    

    NSDate *dt2 = [[NSDate alloc] init];

    dt2 = [df dateFromString:date02];

    /**

     比较两日期时间相差

     */

    double timeDiff = [dt1 timeIntervalSinceDate:dt2 ];

    

    return timeDiff;

}
