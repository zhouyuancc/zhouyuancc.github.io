/**

 xib执行完毕后 调用

 完成了连线操作，self.centerWheel.frame有值

 */

- (void)awakeFromNib

{

    //imageView默认与用户交互为NO

    self.centerWheel.userInteractionEnabled = YES;

    

    /**

     加载大图片

     */

    UIImage *bigImage = [UIImage imageNamed:@"LuckyAstrology"];

    

     /**

     点 : 1*1

     像素 : 2*2

     */

    CGFloat smallW = bigImage.size.width/12;

    CGFloat smallH = bigImage.size.height;//点

    

    /**

     点 -> 像素

     */

//    smallW = smallW * 2;

//    smallH = smallH * 2;

    /**

     判断retina屏幕

     */

    //屏幕的伸缩属性[UIScreen mainScreen].scale

    //retina 2//非retina 1

    smallW = smallW * [UIScreen mainScreen].scale;

    smallH = smallH * [UIScreen mainScreen].scale;

    

    //添加12个按钮

    for(int index = 0 ; index < 12 ; index++)

    {

        UIButton *btn = [UIButton buttonWithType:UIButtonTypeCustom];

        

        /**

         从大图片中裁剪对应星座的图片

         CGImageRef中的rect使用的是 像素

         */

        CGFloat smallX = index * smallW;

        CGRect smallRect = CGRectMake(smallX, 0, smallW, smallH);

        

/*********************************************************************/

        //从 左边大图片 裁剪 右边矩形

        CGImageRef smallImage = CGImageCreateWithImageInRect(bigImage.CGImage, smallRect);//(像素,像素)

/*********************************************************************/

        

//        btn.backgroundColor = [UIColor colorWithRed:arc4random_uniform(255)/255.0 green:arc4random_uniform(255)/255.0 blue:arc4random_uniform(255)/255.0 alpha:1.0];

        

        [btn setImage:[UIImage imageWithCGImage:smallImage] forState:UIControlStateNormal];

        

        [btn setBackgroundImage:[UIImage imageNamed:@"LuckyRototeSelected"] forState:UIControlStateSelected];

        

        btn.bounds = CGRectMake(0, 0, 68, 143);

        

        /**

         设置锚点和位置

         */

        //锚点

        btn.layer.position = CGPointMake(self.centerWheel.frame.size.width * 0.5, self.centerWheel.frame.size.height * 0.5);

        btn.layer.anchorPoint = CGPointMake(0.5, 1);

        

        /**

         设置旋转角度

         */

        //需要旋转的角度

        CGFloat angle = (30 * index)/180.0 * M_PI;

        //绕锚点旋转

        btn.transform = CGAffineTransformMakeRotation(angle);

        

        //监听按钮点击

        [btn addTarget:self action:@selector(btnClick:) forControlEvents:UIControlEventTouchUpInside];

        

        [self.centerWheel addSubview:btn];

        

//        btn.selected = YES;

    }

}
