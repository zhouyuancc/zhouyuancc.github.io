UIImageView *imageV = [[UIImageView alloc] init];

——————————————————————————————————————————————————————————————————————————————————————————————————————

        /**

         UIViewContentModeScaleToFill 填充整个控件

         

         UIViewContentModeScaleAspectFit 按比例缩放，并且宽高！都不！超出控件

         UIViewContentModeScaleAspectFill 按比例缩放，直到！有一个！宽度或高度不超过控件就可以，自动调节中心点

         

         UIViewContentModeCenter 不会缩放图片，只会把图片的中心点对齐控件

         */

        imageV.contentMode = UIViewContentModeScaleAspectFill;

        

        //裁剪图片，超出控件的部分裁剪掉

        imageV.clipsToBounds = YES;
