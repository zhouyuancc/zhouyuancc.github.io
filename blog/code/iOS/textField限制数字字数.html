    //监听文本框文字发生改变通知

    [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(textChange) name:UITextFieldTextDidChangeNotification object:self.password];

    

    [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(textChange) name:UITextFieldTextDidChangeNotification object:self.validate];



/**

 文本框的文字发生改变的时候调用

 限制数字字数

 */

- (void)textChange

{

    

    if (self.password.text.length > 6) {

        [MBProgressHUD showError:@"超过最大字数"];

        self.password.text = [self.password.text substringToIndex:6];

        return;

    }

    if (self.validate.text.length > 6) {

        [MBProgressHUD showError:@"超过最大字数"];

        self.validate.text = [self.validate.text substringToIndex:6];

        return;

    }

}


- (void)dealloc

{

    //必须 移除监听

    [[NSNotificationCenter defaultCenter] removeObserver:self];

}
