- (NSArray *)foodTypes

{

    if (_foodTypes == nil) {

        

        //从plist文件中获得模型

        

        //法一:获得数组

        NSArray *dictArray = [NSArray arrayWithContentsOfFile:[[NSBundle mainBundle] pathForResource:@"food_types.plist" ofType:nil]];

        //数组转模型

        _foodTypes = [FoodType objectArrayWithKeyValuesArray:dictArray];

        

        //法二:plist转模型

        _foodTypes = [FoodType objectArrayWithFilename:@"food_types.plist"];

        

    }

    return _foodTypes;

}
