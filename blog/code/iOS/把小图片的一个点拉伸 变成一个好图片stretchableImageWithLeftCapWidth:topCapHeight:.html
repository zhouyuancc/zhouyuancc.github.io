@property (weak, nonatomic) IBOutlet UIButton *loginButton;



- (void)viewDidLoad

{

    [super viewDidLoad];

    

    UIImage *image = [UIImage imageNamed:@"RedButton"];

    UIImage *imageHigh = [UIImage imageNamed:@"RedButtonPressed"];


    //把小图片的一个点拉伸 变成一个好图片

    image = [image stretchableImageWithLeftCapWidth:image.size.width * 0.5 topCapHeight:image.size.height * 0.5];

    

    imageHigh = [imageHigh stretchableImageWithLeftCapWidth:imageHigh.size.width * 0.5 topCapHeight:imageHigh.size.height * 0.5];

    

//    [image resizableImageWithCapInsets:<#(UIEdgeInsets)#>];

//    [image resizableImageWithCapInsets:<#(UIEdgeInsets)#> resizingMode:<#(UIImageResizingMode)#>];

    

    [self.loginButton setBackgroundImage:image forState:UIControlStateNormal];

    

    [self.loginButton setBackgroundImage:imageHigh forState:UIControlStateHighlighted];

    

}
