@property (nonatomic,weak) UIView *divider;

/**

 如果cell是通过 storyboard或者xib 创建的，就不可能会调用这个方法来初始化cell

 如果cell是通过 手写代码 创建，会调用initWithStyle：reuseIdentifier：来初始化cell

 这个方法的作用类似于init方法

 */

- (void)awakeFromNib {//从storyboard中生成就会调用awakeFromNib//storyboard是从xib进化来的

    // Initialization code初始化代码

    

    //添加分割线

    UIView *divider = [[UIView alloc] init];

    divider.backgroundColor = [UIColor blackColor];

    [self.contentView addSubview:divider];

    divider.alpha = 0.1;

    self.divider = divider;

}


/**

 设置子控件的frame

 //最真实的高度

 */

- (void)layoutSubviews

{

    [super layoutSubviews];

    

    //设置子控件的frame

    CGFloat dividerX= 0;

    CGFloat dividerH= 1;

    CGFloat dividerY= self.frame.size.height - dividerH;

    CGFloat dividerW= self.frame.size.width;

    

    self.divider.frame = CGRectMake(dividerX, dividerY, dividerW, dividerH);


}
