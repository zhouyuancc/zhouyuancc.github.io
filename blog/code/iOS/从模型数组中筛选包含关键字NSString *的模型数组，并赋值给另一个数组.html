//重写搜索方法

-(NSPredicate *)setSearchWay:(NSString *)changeinfo

{

//参数Run_name、bzlc是模型WorkInfo的成员变量

    return [NSPredicate predicateWithFormat:@"self.%@ contains[cd] %@ OR self.%@ contains[cd] %@", @"Run_name", changeinfo, @"bzlc", changeinfo];


}
