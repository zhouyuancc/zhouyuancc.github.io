@interface LoginViewController ()<Input_OnlyText_CellDelegate,UIActionSheetDelegate>

@property (nonatomic,copy) NSString *netName;

@end


#pragma mark - Input_OnlyText_CellDelegate方法

//点击常用网络后 调用

- (void)Input_OnlyText_CellBtnChangeNet

{    

    NSMutableArray *array = [[NetworkAppdataStore sharedStore] loadAllItems];

    Network *net0 = array[0];//网络1数据

    Network *net1 = array[1];//网络2数据

    

    UIActionSheet *sheet = [[UIActionSheet alloc]initWithTitle:nil delegate:self cancelButtonTitle:@"取消" destructiveButtonTitle:nil otherButtonTitles:net0.name, net1.name, nil];

    [sheet showInView:self.view];

}


#pragma mark - actionSheet的代理方法

- (void)actionSheet:(UIActionSheet *)actionSheet clickedButtonAtIndex:(NSInteger)buttonIndex

{

//    NSLog(@"actionSheet %@ , buttonIndex %ld ",actionSheet,(long)buttonIndex);

    

    if(buttonIndex == 2) return;

    

    if(buttonIndex == 0 || 1)//网络1||2

    {

        /**

         通过buttonIndex获取网络名称

         */

        self.netName = [actionSheet buttonTitleAtIndex:buttonIndex];

        NSLog(@"netName %@ ",self.netName);

    }

}
