//  函数式编程思想


#import "ViewController.h"


#import "Caculator.h"


@interface ViewController ()


@end


@implementation ViewController


// 什么是函数式思想:把一个操作写成一系列的方法.

// 需求:写一个加法计算器,并且加法计算器自带判断是否等于某个值.

- (void)viewDidLoad {

    [super viewDidLoad];

    // Do any additional setup after loading the view, typically from a nib.

    Caculator *caculator = [[Caculator alloc] init];

    

    // 10 + 20 + 30 + 40

//    [[[[caculator add:10] add:20] add:30] add:40];

   BOOL isEqule = [[[caculator add:^(int result){

       // 把计算的事情写到Block

        //二

        result += 10;

        result += 20;

        result += 30;

        result += 40;

        

        return result;

        

    }] equle:^BOOL(int result) {//四

      

        //六

        return result == 100;

    }] isEqule];//八

    

    NSLog(@"%d",isEqule);

}


- (void)didReceiveMemoryWarning {

    [super didReceiveMemoryWarning];

    // Dispose of any resources that can be recreated.

}


@end

- (instancetype)add:(int (^)(int result))block

{

    //一

    _result = block(_result);

    

    //三

    return self;

}


- (instancetype)equle:(BOOL (^)(int))block

{

    //五

    _isEqule = block(_result);

    

    //七

    return self;

}

@end
