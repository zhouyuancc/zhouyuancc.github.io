nil属性

//nonnull不能为空，setter和getter都不能为nil

@property (nonatomic, strong,nonnull) NSArray * names;

@property (nonatomic, strong) NSArray * __nonnull names;


// null_resettable : setter可以为nil, getter不可以为nil

@property (null_resettable, nonatomic, strong) NSArray *names;


    begin和end之间都默认是nonnull不为空

// NS_ASSUME_NONNULL_BEGIN和NS_ASSUME_NONNULL_END之间的所有属性默认都是nonnull

NS_ASSUME_NONNULL_BEGIN


// 默认都是nonnull

@property (nonatomic, strong) NSArray *a1;

@property (nonatomic, strong) NSArray *a2;

@property (nonatomic, strong) NSArray *a3;

@property (nonatomic, strong) NSArray *a4;

@property (nonatomic, strong) NSArray *a5;


NS_ASSUME_NONNULL_END


    KVC 字典赋值、取值

[self.books setObject:<#(nonnull id)#> forKey:<#(nonnull id<NSCopying>)#>];

等价于 self.books[] = ;

    

[self.books objectForKey:<#(nonnull id)#>];

等价于 self.books[];

    __covariant（协变性） __contravariant（逆变性）

    XMGBag *bag0 = nil;

    XMGBag<NSString *> *bag1 = nil;

    XMGBag<NSMutableString *> *bag2 = nil;

    

    bag0 = bag1;

    bag0 = bag2;

    

    bag1 = bag0;

    bag2 = bag0;

    

//    bag1 = (XMGBag<NSString *> *)bag2;

//    bag2 = (XMGBag<NSMutableString *> *)bag1;

    bag1 = bag2; // __covariant（协变性）: 小类型(泛型类的子类类型) -> 大类型(泛型类的父类类型)

    bag2 = bag1; // __contravariant（逆变性）: 大类型(泛型类的父类类型) -> 小类型(泛型类的子类类型)
